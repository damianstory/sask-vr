**Tile toggle logic:**
```typescript
const handleTileToggle = (tileId: string) => {
  if (selectedTiles.includes(tileId)) {
    // Deselect
    setSelectedTiles(selectedTiles.filter(id => id !== tileId))
  } else if (selectedTiles.length >= data.maxSelections) {
    // At max -- shake + message
    setShakeId(tileId)
    setOverflowMessage(true)
    setTimeout(() => setShakeId(null), 300)
    setTimeout(() => setOverflowMessage(false), 3000)
  } else {
    // Select
    setSelectedTiles([...selectedTiles, tileId])
  }
}
```

**Continue button label logic:**
```typescript
const buttonLabel = selectedTiles.length === 0
  ? 'Pick at least 2'
  : selectedTiles.length === 1
    ? 'Pick 1 more'
    : 'Continue \u2192'
const isDisabled = selectedTiles.length < data.minSelections
```

Note: The Continue button is visual only in this component -- actual navigation is handled by the parent Navigation component. The button communicates selection status.

      <span className="text-[14px] font-[800]">{index + 1}</span>
    </button>
